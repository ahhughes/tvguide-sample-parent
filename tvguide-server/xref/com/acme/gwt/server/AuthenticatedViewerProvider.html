17  package com.acme.gwt.server;
18  
19  import javax.persistence.EntityManager;
20  
21  import com.acme.gwt.data.TvViewer;
22  import com.google.inject.Inject;
23  import com.google.inject.Provider;
24  
25  /**
26   * Stores the current user. Currently it keeps a ref to the user object, but this might cause 
27   * issues, in which case it will probably just keep the id, and get the user each time it is asked.
28   * 
29   * Any other auth logic or tracking can go here as well, and it might be wise to push the login
30   * code here too, or at least let this class deal with it, so that something like OpenID can be
31   * wired in easily.
32   * 
33   * As it is now, this MUST be correctly scoped in whatever container it is used in.
34   * 
35   * @author colin
36   *
37   */
38  public class AuthenticatedViewerProvider implements Provider<TvViewer> {
39  	private Provider<EntityManager> emProvider;
40  
41  	// instead of this, ask for generic session provider, where the session just holds an ID
42  	// with which to find the current viewer. Servlet impl will just be session scoped
43  	private Provider<SessionProvider> session;
44  
45  	@Inject
46  	public AuthenticatedViewerProvider(Provider<EntityManager> emp,
47  			Provider<SessionProvider> sessionProvider) {
48  		this.emProvider = emp;
49  		this.session = sessionProvider;
50  	}
51  
52  	@Override
53  	public TvViewer get() {
54  		return session.get().get() == null ? null : emProvider.get().find(
55  				TvViewer.class, session.get().get());
56  	}
57  	public void setCurrentViewer(TvViewer viewer) {
58  		session.get().setActiveViewerId(viewer == null ? null : viewer.getId());
59  	}
60  
61  	public interface SessionProvider extends Provider<Long> {
62  		void setActiveViewerId(Long viewerId);
63  	}
64  }
